53 void loadParamsFromXml(xmlNodePtr node, std::map<std::string, MatNd*>& parameters)
54 {
55  // lookup nodes for every parameter
56  for (auto& entry : parameters) {
57  xmlNodePtr paramNode = getXMLChildByName(node, entry.first.c_str());
58 
59  if (paramNode != NULL) {
60  MatNd* paramStore = entry.second;
61  // verify parameter shape
62  if (paramStore->n == 1) {
63  // 1d column array
64  unsigned int shape;
65  RCHECK(getXMLNodePropertyUnsignedInt(paramNode, "shape", &shape));
66  RCHECK_EQ(shape, paramStore->m);
67  }
68  else if (paramStore->m == 1) {
69  // 1d row array
70  unsigned int shape;
71  RCHECK(getXMLNodePropertyUnsignedInt(paramNode, "shape", &shape));
72  RCHECK_EQ(shape, paramStore->n);
73  }
74  else {
75  // 2d array
76  unsigned int shape[2];
77  RCHECK(getXMLNodePropertyUnsignedIntN(paramNode, "shape", shape, 2));
78  RCHECK_EQ(shape[0], paramStore->m);
79  RCHECK_EQ(shape[1], paramStore->n);
80  }
81 
82  // load parameter values
83  //RCHECK(getXMLNodeVecN(paramNode, paramStore->ele, paramStore->m * paramStore->n));
84  // unfortunately, getXMLNodeVecN only supports strings of up to 512 bytes, but a large matrix can be much larger
85  // thus we need custom code
86 
87  // get content from node
88  xmlChar* txt = xmlNodeGetContent(paramNode);
89  RCHECK(txt);
90  // copy content into an std::string
91  std::string content((char*) txt);
92  // free data from xml
93  xmlFree(txt);
94 
95  // put it into a stringstream
96  std::istringstream is(content);
97  // be sure to use C locale
98  is.imbue(std::locale("C"));
99 
100  // iterate over elements
101  unsigned int nEle = paramStore->m*paramStore->n;
102  for (unsigned int i = 0; i < nEle; ++i) {
103  // read it from stream
104  if (!(is >> paramStore->ele[i])) {
105  RFATAL("Invalid matrix entry at (%d, %d)", i/paramStore->n, i%paramStore->n);
106  }
107  }
108 
109  //MatNd_printComment(entry.first.c_str(), paramStore);
110  }
111  }
112 }

114 static MLPPolicy::Nonlinearity loadNonlinearity(xmlNodePtr node, const char* key)
115 {
116  std::string functionName;
117  if (!getXMLNodePropertySTLString(node, key, functionName)) {
118  return NULL;
119  }
120 
121  // check function name
122  if (functionName == "tanh") {
123  return MatNd_tanh_inplace;
124  }
125  else {
126  RFATAL("Unsupported nonlinearity '%s' for %s", functionName.c_str(), key);
127  }
128 }

130 MLPPolicy* loadMLPPolicyFromXmlNode(
131  xmlNodePtr node,
132  unsigned int input_size,
133  unsigned int output_size)
134 {
135  // load policy description
136  std::vector<std::string> hidden_size_str;
137  getXMLNodePropertyVecSTLString(node, "hidden_sizes", hidden_size_str);
138  std::vector<unsigned int> hidden_sizes;
139  for (auto& str : hidden_size_str) {
140  std::istringstream is(str);
141  is.imbue(std::locale("C"));
142 
143  unsigned int size;
144  if (!(is >> size)) {
145  // invalid format
146  RFATAL("Invalid hidden size entry: %s", str.c_str());
147  }
148  hidden_sizes.push_back(size);
149  }
150 
151  // load nonlinearities
152  MLPPolicy::Nonlinearity hidden_nl = loadNonlinearity(node, "hidden_nonlinearity");
153  MLPPolicy::Nonlinearity output_nl = loadNonlinearity(node, "output_nonlinearity");
154 
155  // create MLP
156  MLPPolicy* mlp = new MLPPolicy(input_size, output_size, hidden_sizes, hidden_nl, output_nl);
157 
158  // load parameters
159  auto params = mlp->getParameters();
160  loadParamsFromXml(node, params);
161 
162  return mlp;
163 }

165 MLPPolicy* loadMLPPolicyFromXml(const char* xmlFile, unsigned int input_size, unsigned int output_size)
166 {
167  // Determine absolute file name of config file and copy the XML file name
168  char filename[256] = "";
169  bool fileExists = Rcs_getAbsoluteFileName(xmlFile, filename);
170 
171  if (!fileExists) {
172  RMSG("Resource path is:");
173  Rcs_printResourcePath();
174  RFATAL("Experiment configuration file \"%s\" not found in "
175  "ressource path - exiting", xmlFile ? xmlFile : "NULL");
176  }
177 
178  // load xml tree
179  xmlDocPtr doc;
180  xmlNodePtr node = parseXMLFile(filename, "MLPPolicy", &doc);
181  RCHECK(node);
182 
183  // create MLP
184  MLPPolicy* mlp = loadMLPPolicyFromXmlNode(node, input_size, output_size);
185 
186  // free xml tree
187  xmlFreeDoc(doc);
188 
189  return mlp;
190 }

220 void MLPPolicy::MLPLayer::compute(MatNd* output, const MatNd* input)
221 {
222  // o = i * W + b
223  MatNd_mul(output, input, weights);
224  MatNd_addSelf(output, bias);
225  // o = NL(o)
226  if (nonlinearity != NULL) {
227  nonlinearity(output);
228  }
229 }

262 MLPPolicy::MLPPolicy(
263  unsigned int input_size,
264  unsigned int output_size,
265  std::vector<unsigned int> hidden_sizes,
266  MLPPolicy::Nonlinearity hidden_nonlinearity,
267  MLPPolicy::Nonlinearity output_nonlinearity)
268 {
269  unsigned int next_layer_input_size = input_size;
270  // build hidden layers
271  hiddenLayers.resize(hidden_sizes.size());
272  for (size_t i = 0; i < hidden_sizes.size(); ++i) {
273  unsigned int layerWidth = hidden_sizes[i];
274  // create layer
275  hiddenLayers[i].init(next_layer_input_size, layerWidth, hidden_nonlinearity);
276  // next layer input size is current layer output size
277  next_layer_input_size = layerWidth;
278  }
279  // build output layer
280  outputLayer.init(next_layer_input_size, output_size, output_nonlinearity);
281 
282  inputSize = input_size;
283  outputSize = output_size;
284 }

288 void MLPPolicy::computeAction(MatNd* action, const MatNd* observation)
289 {
290  // in rllab, vectors are column vectors. For the MLP formula from rllab, we need them as row vectors
291  // luckily, transposing vectors is easy
292  RCHECK(observation->n == 1);
293  MatNd inputTransposed = MatNd_fromPtr(1, observation->m, observation->ele);
294  RCHECK(action->n == 1);
295  MatNd outputTransposed = MatNd_fromPtr(1, action->m, action->ele);
296 
297  // points to the input of the next layer, always non-owning
298  const MatNd* nextLayerInput = &inputTransposed;
299  // compute hidden layer values
300  for (auto& layer : hiddenLayers) {
301  layer.compute(layer.outputCache, nextLayerInput);
302  // output -> next layer input
303  nextLayerInput = layer.outputCache;
304  }
305  // compute output layer values
306  outputLayer.compute(&outputTransposed, nextLayerInput);
307 }
