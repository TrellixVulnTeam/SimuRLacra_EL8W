64  virtual ActionModel* createActionModel()
65  {
66  // Setup inner action model
67  RcsBody* left = RcsGraph_getBodyByName(graph, "PowerGrasp_L");
68  RCHECK(left);
69  RcsBody* right = RcsGraph_getBodyByName(graph, "PowerGrasp_R");
70  RCHECK(right);
71 
72  // Control effector positions (not orientation)
73  std::unique_ptr<AMIKGeneric> innerAM(new AMIKGeneric(graph));
74  innerAM->addTask(new TaskPosition3D(graph, left, nullptr, nullptr));
75  innerAM->addTask(new TaskPosition3D(graph, right, nullptr, nullptr));
76 
77  // Incorporate collision costs into IK
78  if (properties->getPropertyBool("collisionAvoidanceIK", true)) {
79  REXEC(4) {
80  std::cout << "IK considers the provided collision model" << std::endl;
81  }
82  innerAM->setupCollisionModel(collisionMdl);
83  }
84 
85  // Obtain task data
86  std::vector<std::unique_ptr<DynamicalSystem>> tasks;
87  auto& tsLeft = properties->getChildList("tasksLeft");
88  for (auto tsk : tsLeft) {
89  DynamicalSystem* ds = DynamicalSystem::create(tsk, 3);
90  tasks.emplace_back(new DSSlice(ds, 0, 3));
91  }
92  auto& tsRight = properties->getChildList("tasksRight");
93  for (auto tsk : tsRight) {
94  DynamicalSystem* ds = DynamicalSystem::create(tsk, 3);
95  tasks.emplace_back(new DSSlice(ds, 3, 3));
96  }
97  if (tasks.empty()) {
98  throw std::invalid_argument("No tasks specified!");
99  }
100 
101  // Setup task-based action model
102  std::vector<DynamicalSystem*> taskRel;
103  for (auto& task : tasks) {
104  taskRel.push_back(task.release());
105  }
106 
107  // Get the method how to combine the movement primitives / tasks given their activation
108  std::string taskCombinationMethod = "unspecified";
109  properties->getProperty(taskCombinationMethod, "taskCombinationMethod");
110  TaskCombinationMethod tcm = AMDynamicalSystemActivation::checkTaskCombinationMethod(taskCombinationMethod);
111 
112  return new AMDynamicalSystemActivation(innerAM.release(), taskRel, tcm);
113  }
114 
115  virtual ObservationModel* createObservationModel()
116  {
117  // Observe effector positions
118  std::unique_ptr<OMCombined> fullState(new OMCombined());
119 
120  auto left = new OMBodyStateLinear(graph, "PowerGrasp_L");
121  fullState->addPart(left);
122 
123  auto right = new OMBodyStateLinear(graph, "PowerGrasp_R");
124  fullState->addPart(right);
125 
126  auto amAct = actionModel->unwrap<AMDynamicalSystemActivation>();
127  RCHECK(amAct);
128  fullState->addPart(new OMDynamicalSystemGoalDistance(amAct));
129 
130  if (properties->getPropertyBool("observeCollisionCost", true) & (collisionMdl != nullptr)) {
131  // Add the collision cost observation model
132  auto omColl = new OMCollisionCost(collisionMdl);
133  fullState->addPart(omColl);
134  }
135 
136  if (properties->getPropertyBool("observePredictedCollisionCost", false) & (collisionMdl != nullptr)) {
137  // Get the horizon from config
138  int horizon = 20;
139  properties->getChild("collisionConfig")->getProperty(horizon, "predCollHorizon");
140  // Add the collision cost observation model
141  auto omCollPred = new OMCollisionCostPrediction(graph, collisionMdl, actionModel, horizon);
142  fullState->addPart(omCollPred);
143  }
144 
145 
146  return fullState.release();
147  }

151  getHUDText(

153  PhysicsBase* simulator, PhysicsParameterManager* physicsManager, ForceDisturber* forceDisturber) override

165  unsigned int numPosCtrlJoints = 0;

167  // Iterate over unconstrained joints
168  RCSGRAPH_TRAVERSE_JOINTS(graph) {

171  numPosCtrlJoints++;
172  }

174  numTrqCtrlJoints++;
175  }
176  }
177  }
178  linesOut.emplace_back(
179  string_format("num joints: %d total, %d pos ctrl, %d trq ctrl", graph->nJ, numPosCtrlJoints,
180  numTrqCtrlJoints));
181 
182  linesOut.emplace_back(string_format(
183  " left hand pg: [% 3.2f,% 3.2f,% 3.2f] m",
184  obs->ele[0], obs->ele[1], obs->ele[2]));
185 
186  linesOut.emplace_back(string_format(
187  "right hand pg: [% 3.2f,% 3.2f,% 3.2f] m",
188  obs->ele[3], obs->ele[4], obs->ele[5]));
189 
190  linesOut.emplace_back(string_format(
191  "goal distance: [% 3.2f,% 3.2f] m",
192  obs->ele[6], obs->ele[7]));
193 
194  auto omColl = observationModel->findOffsets<OMCollisionCost>();
195  if (omColl) {
196  linesOut.emplace_back(string_format(
197  "collision cost: % 3.2f",
198  obs->ele[omColl.pos]));
199  }
200 
201  auto omCollPred = observationModel->findOffsets<OMCollisionCostPrediction>();
202  if (omCollPred) {
203  linesOut.emplace_back(string_format(
204  "collision cost (pred): % 3.2f",
205  obs->ele[omCollPred.pos]));
206  }
207  }
